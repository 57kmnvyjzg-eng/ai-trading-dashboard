// 모델명을 가장 확실한 gemini-1.5-pro-latest로 교체하고 에러 상세 추적 기능 추가 async function runGeminiAnalysis() { const token = localStorage.getItem('gemini_api_key'); if(!token) { alert("왼쪽 아래 [⚙️ API 설정]에서 AI Studio 토큰을 먼저 입력해주세요!"); openApiModal(); return; } const ticker = document.getElementById('p3Ticker').value.trim() || 'NVDA'; const btn = document.getElementById('btnRunAi'); const container = document.getElementById('main_chart_container'); btn.innerHTML = `
분석 중...`; btn.disabled = true; container.innerHTML = `
AI가 ${ticker} 차트 궤적을 연산하고 있습니다...`; const prompt = `너는 월스트리트 퀀트 AI야. '${ticker}' 주식의 가상 일봉 20개(baseData)와 미래 5일 시나리오 3개(planA, planB, planC)의 lineData, ghostCandles를 오직 JSON 포맷으로만 생성해줘. 마크다운 백틱이나 다른 텍스트는 일체 쓰지 마. 형식: {"baseData":[{"time":1715000000,"open":120,"high":122,"low":118,"close":121}], "planA":{"lineData":[{"time":1716728000,"value":121}],"ghostCandles":[{"time":1716728000,"open":121,"high":123,"low":120,"close":122}]}, "planB":{"lineData":[],"ghostCandles":[]}, "planC":{"lineData":[],"ghostCandles":[]}}`; try { // 수정 포인트: 모델 이름을 gemini-1.5-pro-latest 로 변경 const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-pro-latest:generateContent?key=${token}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { temperature: 0.2 } }) }); if(!response.ok) { // 구글 서버가 뱉어내는 진짜 에러 메시지를 화면에 띄웁니다. const errorData = await response.json().catch(() => ({})); const exactReason = errorData.error?.message || "알 수 없는 오류"; throw new Error(`Google API 거절 사유: ${exactReason}`); } const resData = await response.json(); let aiText = resData.candidates[0].content.parts[0].text; aiText = aiText.replace(/```json/gi, '').replace(/```/g, '').trim(); const parsedData = JSON.parse(aiText); drawChart(parsedData); document.getElementById('chartTitle').textContent = `📈 ${ticker} AI 실시간 궤적 예측`; document.getElementById('chartLegends').style.display = 'flex'; } catch (error) { console.error(error); // 에러 상세 내용을 붉은 글씨로 화면에 크게 출력 container.innerHTML = `
통신 실패!
${error.message}
`; } finally { btn.innerHTML = `⚡ AI 실시간 분석`; btn.disabled = false; } }
